<stv-bar-chart>

Demonstrating a bar chart with stock ticker symbols / prices.

        <stv-bar-chart
          chart-id="myStvBarChart"
          legend-metric="name"
          linear-metric="value"
          linear-tick-format="USD"
          margin-left="35"
          ordinal-metric="ticker"
          responsive
          tooltips
          x-label="Ticker Symbol"
          y-label="Price"
        ></stv-bar-chart>

        ...

        document.addEventListener('stv-bar-chart-loaded', function() {
          document.querySelector('stv-bar-chart').chartData = [...your data...]
        })
